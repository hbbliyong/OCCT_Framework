set(LIBARAY_NAME Sample_App)
include_directories(
    ${VTK_INCLUDE_DIRS}
)

# 1. 先递归收集所有源文件（.c/.cpp 等）
file(GLOB_RECURSE SOURCE_FILES CONFIGURE_DEPENDS 
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.c*"
)
file(GLOB_RECURSE HEADER_FILES CONFIGURE_DEPENDS 
    "${CMAKE_CURRENT_SOURCE_DIR}/includes/*.h*"
)
# 2. 按原始目录结构分组（VS 中显示为 src/xxx/xxx.cpp）
source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}/src" 
            PREFIX "Source Files" 
            FILES ${SOURCE_FILES})

# 3. 头文件同理

source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}/includes" 
            PREFIX "Header Files" 
            FILES ${HEADER_FILES})

add_executable(${LIBARAY_NAME}
    ${SOURCE_FILES}
    ${HEADER_FILES}
)
target_include_directories(${LIBARAY_NAME} PRIVATE includes)
target_link_libraries(${LIBARAY_NAME}
    Qt6::Widgets
    ${VTK_LIBRARIES}
    ${OpenCASCADE_LIBRARIES}
    OCCT_Framework
)
